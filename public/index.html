shaderup

Go from GLSL to live WebGL in seconds.

<!-- The magic behind this page's background -->
<script type="module">
  import { ShaderUp } from "https://unpkg.com/shaderup@0.3.0";

  const shader = new ShaderUp({
    fragmentShader: await fetch("./landing.frag").then(r => r.text())
  });
  
  shader.start();
</script>
